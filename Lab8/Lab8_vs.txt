#version 300 es

layout (location = 0) in vec4 aPosition;
layout (location = 2) in vec2 aTexcoord;

uniform mat4 uModelMat, uViewMat, uProjMat;

out vec2 vTexcoord;

void main()
{
	mat4 modelViewProjMat = uProjMat * uViewMat * uModelMat;
	gl_Position = modelViewProjMat * aPosition;
	vTexcoord = aTexcoord;
}
