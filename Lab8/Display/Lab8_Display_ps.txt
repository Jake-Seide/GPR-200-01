#version 300 es

#ifdef GL_ES
precision highp float;
#endif // GL_ES

layout (location = 0) out vec4 rtFragColor;

in vec2 vTexcoord;

uniform sampler2D otherTex;


void main()
{
	vec2 uv = vTexcoord;
	vec4 col = texture(otherTex, uv);
	rtFragColor = col;
	
	//rtFragColor = vec4(1.5, 0.5, 0.5, 1.0);
	//rtFragColor = texture(uTex, vTexcoord);
}
