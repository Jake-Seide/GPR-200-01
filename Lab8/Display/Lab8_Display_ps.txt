#version 300 es

#ifdef GL_ES
precision highp float;
#endif // GL_ES

layout (location = 0) out vec4 rtFragColor;

//Varying(ps)
in vec2 vTexcoord;

// Texture Uniforms
uniform sampler2D uTex_0;
uniform sampler2D uTex_1;

// Combining Scene Pass and Blurr Pass to create final display
vec4 ImageComposite()
{
	vec4 val = texture(uTex_1, vTexcoord); // Scene Pass
	vec4 sum = texture(uTex_0, vTexcoord); // Blurr Pass
	
	return sum + val;
}

void main()
{
	// Calling Final Display
	rtFragColor = ImageComposite();
}
